     ```bash
     npm install -D tailwindcss @tailwindcss/vite
     ```
  4. Update `vite.config.ts` to include the Tailwind CSS Vite plugin:
     ```typescript
     import { defineConfig } from 'vite'
     import react from '@vitejs/plugin-react'
     import tailwindcss from '@tailwindcss/vite'

     export default defineConfig({
       plugins: [
         react(),
         tailwindcss(),
       ],
     })
     ```
  5. Replace `src/index.css` content with Tailwind CSS 4 import:
     ```css
     @import "tailwindcss";
     ```

  9. Delete any default Vite boilerplate files that are not needed: `src/App.css`, `src/assets/react.svg`, `public/vite.svg`.
